### 2. 在奖励模型中使用场景信息

**文件: `RL/verl/utils/reward_score/custom_reward_with_scenes.py`**（新增）

```python
import torch
import numpy as np
from typing import List, Tuple

class SceneAwareRewardModel:
    """计算每个场景分数的奖励模型。"""

    def __init__(self, base_reward_fn):
        self.base_reward_fn = base_reward_fn

    def compute_reward_with_scenes(
        self,
        predictions,
        ground_truth,
        scene_boundaries: List[Tuple[int, int]],
    ) -> torch.Tensor:
        """
        计算按场景分组的奖励。

        Args:
            predictions: 模型输出 (B, T, D)
            ground_truth: 真实标签 (B, T, D)
            scene_boundaries: 每个批次的场景间隔列表 [(start, end), ...]

        Returns:
            奖励 (B,) 具有场景感知加权
        """
        batch_size = predictions.shape[0]
        batch_rewards = []

        for i in range(batch_size):
            pred_i = predictions[i]
            truth_i = ground_truth[i]
            scenes_i = scene_boundaries[i] if isinstance(scene_boundaries[0], list) else scene_boundaries

            scene_scores = []
            for start, end in scenes_i:
                scene_pred = pred_i[start:end]
                scene_truth = truth_i[start:end]

                # 计算此场景的基础奖励
                scene_reward = self.base_reward_fn(scene_pred, scene_truth)
                scene_scores.append(scene_reward)

            # 聚合场景奖励
            batch_reward = torch.mean(torch.tensor(scene_scores))
            batch_rewards.append(batch_reward)

        return torch.tensor(batch_rewards)

    def __call__(self, batch):
        predictions = batch['predictions']
        ground_truth = batch['ground_truth']
        scene_boundaries = batch.get('scene_boundaries', None)

        if scene_boundaries is None:
            # 回退到基础奖励模型
            return self.base_reward_fn(predictions, ground_truth)

        return self.compute_reward_with_scenes(
            predictions,
            ground_truth,
            scene_boundaries
        )
```

### 3. 在训练期间访问场景信息

**文件: `RL/verl/trainer/ppo/ray_trainer.py`**（修改rollout循环）

```python
def compute_rewards(self, rollout_batch):
    """使用场景感知计算奖励。"""

    # 从批次提取场景边界
    scene_boundaries = rollout_batch.get('scene_boundaries', None)

    # 如果可用场景边界，使用场景感知奖励计算
    if scene_boundaries is not None:
        rewards = self.scene_aware_reward_fn(rollout_batch)
    else:
        # 回退到标准奖励计算
        rewards = self.reward_fn(rollout_batch)

    return rewards
```

def preprocess_function(examples):
    """使用场景感知处理示例。"""

    # 检查场景边界是否可用
    has_scenes = 'scene_boundaries' in examples

    processed = {}
    for key in examples:

### 3. 按场景组织训练

**文件: `SFT/src/llamafactory/data/dataset_merger.py`**（修改collate函数）

```python
def collate_fn_with_scenes(batch):
    """在组织场景的同时整理批次。"""

    processed_batch = {}

    # 如果可用，按场景组织
    has_scenes = any('scene_boundaries' in item for item in batch)

    if has_scenes:
        # 计算每个场景的损失
        scene_data = defaultdict(list)
        for item in batch:
            scenes = item.get('scene_boundaries', [(0, len(item['input_ids']))])
            for scene_start, scene_end in scenes:
                scene_data[f'scene_{len(scene_data)}'].append({
                    'start': scene_start,
                    'end': scene_end,
                    'item': item
                })

        processed_batch['scene_data'] = scene_data

    # 标准整理
    for key in batch[0].keys():
        if key != 'scene_boundaries':
            processed_batch[key] = torch.stack([torch.tensor(item[key]) for item in batch])

    return processed_batch
```

def analyze_scene_detection(videos, scene_boundaries_list):
    """分析检测到的场景的质量。"""

    total_videos = len(videos)
    total_scenes = sum(len(scenes) for scenes in scene_boundaries_list)
    avg_scene_length = np.mean([
        end - start
        for scenes in scene_boundaries_list
        for start, end in scenes
    ])

    print(f"场景检测分析:")
    print(f"  总视频数: {total_videos}")
    print(f"  总场景数: {total_scenes}")
    print(f"  平均场景/视频: {total_scenes / total_videos:.2f}")
    print(f"  平均场景长度: {avg_scene_length:.2f} 帧")

    # 检查分布
    scene_lengths = [
        end - start
        for scenes in scene_boundaries_list
        for start, end in scenes
    ]
    print(f"  场景长度分布:")

## 示例：完整训练脚本

**文件: `RL/examples/train_with_scenes.py`**（新增）

```python
#!/usr/bin/env python3
"""
使用场景基础处理的示例训练脚本。
"""

import os
from verl.utils import hf_tokenizer, hf_processor
from verl.utils.dataset.multimodal_dataset_video import MultiModalDatasetVideo
from verl.trainer.ppo.ray_trainer import RayPPOTrainer

def train_with_scenes(
    model_path: str,
    data_file: str,
    output_dir: str,
    use_scene_detection: bool = True,
):
    """使用场景基础处理训练VideoR模型。"""

    # 初始化模型组件
    tokenizer = hf_tokenizer(model_path)
    processor = hf_processor(model_path)

    # 使用场景检测创建数据集
    train_dataset = MultiModalDatasetVideo(
        data_files=[data_file],
        tokenizer=tokenizer,
        processor=processor,
        use_scene_detection=use_scene_detection,
    )

    print(f"已加载数据集，包含 {len(train_dataset)} 个示例")

    # 检查场景检测
    sample = train_dataset[0]
    if 'scene_boundaries' in sample:
        print(f"场景检测活跃: 第一个视频中检测到 {len(sample['scene_boundaries'])} 个场景")
    else:
        print("场景检测不活跃")

    # 初始化训练器
    trainer = RayPPOTrainer(
        model_name_or_path=model_path,
        train_dataset=train_dataset,
        output_dir=output_dir,
        # ... 其他训练器参数
    )

    # 训练
    trainer.train()

    # 保存模型
    trainer.save_model(os.path.join(output_dir, 'final_model'))


if __name__ == "__main__":
    import argparse

    parser = argparse.ArgumentParser()
    parser.add_argument('--model_path', required=True, help='基础模型的路径')
    parser.add_argument('--data_file', required=True, help='训练数据的路径')
    parser.add_argument('--output_dir', required=True, help='输出目录')
    parser.add_argument('--no-scenes', action='store_true', help='禁用场景检测')

    args = parser.parse_args()

    train_with_scenes(
        model_path=args.model_path,
        data_file=args.data_file,
        output_dir=args.output_dir,
        use_scene_detection=not args.no_scenes,
    )
```

- 场景检测开销约为 5-10%
- 如果更慢，检查场景检测不是每个批次运行
- 应仅在数据集初始化期间运行一次

### 内存问题

- 场景边界仅为 (int, int) 元组列表
- 内存影响可忽略
- 检查总数据集大小
